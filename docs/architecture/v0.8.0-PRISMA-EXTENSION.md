## 新增模型

```prisma
// ==================== Sprint 7: RBAC 权限管理 ====================

model Role {
  id          String          @id @default(cuid())
  name        String
  code        String          @unique
  description String?
  isSystem    Boolean         @default(false)
  isActive    Boolean         @default(true)
  permissions Json            @default("[]")
  createdAt   DateTime        @default(now())
  updatedAt   DateTime        @updatedAt
  users       UserRole[]
  permissions Permission[]

  @@index([code])
  @@index([isActive])
  @@map("roles")
}

model UserRole {
  id        String    @id @default(cuid())
  userId    String
  roleId    String
  createdAt DateTime  @default(now())
  user      User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  role      Role      @relation(fields: [roleId], references: [id], onDelete: Cascade)

  @@unique([userId, roleId])
  @@index([userId])
  @@index([roleId])
  @@map("user_roles")
}

model Permission {
  id          String    @id @default(cuid())
  code        String    @unique
  name        String
  description String?
  module      String
  action      String
  isSystem    Boolean   @default(true)
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt
  roles       Role[]

  @@index([module])
  @@index([action])
  @@map("permissions")
}

// ==================== Sprint 7: 多平台接入 ====================

model Platform {
  id            String            @id @default(cuid())
  name          String
  displayName   String
  apiKey        String?
  apiSecret     String?
  accessToken   String?
  refreshToken  String?
  tokenExpiresAt DateTime?
  merchantId    String?
  shopId        String?
  baseUrl       String?
  status        PlatformStatus    @default(INACTIVE)
  config        Json              @default("{}")
  lastSyncAt    DateTime?
  createdAt     DateTime          @default(now())
  updatedAt     DateTime          @updatedAt
  orders        Order[]
  listings      ProductListing[]
  syncLogs      PlatformSyncLog[]

  @@index([name])
  @@index([status])
  @@map("platforms")
}

model PlatformSyncLog {
  id          String            @id @default(cuid())
  platformId  String
  syncType    SyncType
  status      SyncStatus
  startTime   DateTime
  endTime     DateTime?
  totalCount  Int               @default(0)
  successCount Int              @default(0)
  failedCount Int               @default(0)
  errorMessage String?
  details     Json?
  createdAt   DateTime          @default(now())
  platform    Platform          @relation(fields: [platformId], references: [id], onDelete: Cascade)

  @@index([platformId])
  @@index([syncType])
  @@index([status])
  @@index([startTime])
  @@map("platform_sync_logs")
}

// ==================== Sprint 7: 产品刊登 ====================

model ProductListing {
  id             String            @id @default(cuid())
  productId      String?
  platformId     String
  platformProductId String?
  sku            String
  title          String
  description    String?
  price          Decimal?          @db.Decimal(10, 2)
  currency       String            @default("USD")
  quantity       Int
  status         ListingStatus     @default(DRAFT)
  platformStatus String?
  errorMessage   String?
  publishedAt    DateTime?
  lastSyncedAt   DateTime?
  metadata       Json              @default("{}")
  images         String[]
  createdAt      DateTime          @default(now())
  updatedAt      DateTime          @updatedAt
  product        Product?          @relation(fields: [productId], references: [id])
  platform       Platform          @relation(fields: [platformId], references: [id], onDelete: Cascade)
  history        ProductListingHistory[]

  @@index([productId])
  @@index([platformId])
  @@index([status])
  @@index([sku])
  @@map("product_listings")
}

model ProductListingHistory {
  id            String            @id @default(cuid())
  listingId     String
  action        String
  beforeData    Json?
  afterData     Json?
  result        String
  errorMessage  String?
  operatorId    String?
  createdAt     DateTime          @default(now())
  listing       ProductListing    @relation(fields: [listingId], references: [id], onDelete: Cascade)

  @@index([listingId])
  @@index([action])
  @@index([createdAt])
  @@map("product_listing_history")
}
```

## 修改现有模型

### 1. User 模型修改

```prisma
model User {
  id             String          @id @default(cuid())
  email          String          @unique
  name           String?
  passwordHash   String?
  role           Role?           // 保留向后兼容，不再使用
  avatar         String?
  organizationId String?         // 新增: 组织隔离
  createdAt      DateTime        @default(now())
  updatedAt      DateTime        @updatedAt
  customers      Customer[]      @relation("Owner")
  orders         Order[]         @relation("SalesRep")
  purchaseOrders PurchaseOrder[] @relation("Purchaser")
  suppliers      Supplier[]      @relation("Owner")
  userRoles      UserRole[]       // 新增: 用户角色关联

  @@map("users")
}
```

### 2. Order 模型修改

```prisma
model Order {
  id                String             @id @default(cuid())
  orderNo           String             @unique
  customerId        String
  sourceInquiryId   String?
  sourceQuotationId String?
  platformId        String?            // 新增: 关联平台
  platformOrderNo   String?            // 新增: 平台订单号
  status            OrderStatus        @default(PENDING)
  currency          String             @default("USD")
  exchangeRate      Decimal            @default(1) @db.Decimal(10, 6)
  totalAmount       Decimal            @db.Decimal(12, 2)
  paidAmount        Decimal            @default(0) @db.Decimal(12, 2)
  balanceAmount     Decimal            @default(0) @db.Decimal(12, 2)
  paymentTerms      String?
  paymentDeadline   DateTime?
  deliveryTerms     String?
  deliveryDate      DateTime?
  deliveryDeadline  DateTime?
  shippingAddress   String?
  shippingContact   String?
  shippingPhone     String?
  salesRepId        String?
  approvalStatus    ApprovalStatus     @default(NOT_REQUIRED)
  approvedBy        String?
  approvedAt        DateTime?
  notes             String?
  internalNotes     String?
  attachments       String[]
  confirmedAt       DateTime?
  completedAt       DateTime?
  cancelledAt       DateTime?
  cancelReason      String?
  createdAt         DateTime           @default(now())
  updatedAt         DateTime           @updatedAt
  items             OrderItem[]
  customer          Customer           @relation(fields: [customerId], references: [id])
  salesRep          User?              @relation("SalesRep", fields: [salesRepId], references: [id])
  payments          Payment[]
  productionRecords ProductionRecord[]
  purchaseOrders    PurchaseOrder[]    @relation("SalesOrder")
  qualityChecks     QualityCheck[]
  shipments         Shipment[]
  outboundOrders    OutboundOrder[]    @relation(name: "OrderOutboundOrders")
  platform          Platform?          @relation(fields: [platformId], references: [id]) // 新增

  @@index([customerId, status])
  @@index([salesRepId, createdAt])
  @@index([status, deliveryDate])
  @@index([createdAt])
  @@index([orderNo])
  @@index([platformId])             // 新增
  @@index([platformOrderNo])        // 新增
  @@map("orders")
}
```

model InventoryItem {

### 4. InventoryLog 模型修改

```prisma
model InventoryLog {
  id              String           @id @default(cuid())
  productId       String
  warehouseId     String           // 修改: 改为外键
  inboundOrderId  String?
  outboundOrderId String?
  type            InventoryLogType @default(IN)
  quantity        Int
  beforeQuantity  Int
  afterQuantity   Int
  referenceType   String?
  referenceId     String?
  note            String?
  createdAt       DateTime         @default(now())
  inboundOrder    InboundOrder?    @relation(fields: [inboundOrderId], references: [id])
  outboundOrder   OutboundOrder?   @relation(fields: [outboundOrderId], references: [id])
  product         Product          @relation(fields: [productId], references: [id])
  inventoryItem   InventoryItem?   @relation(fields: [productId, warehouseId], references: [productId, warehouseId])
  warehouse       Warehouse        @relation(fields: [warehouseId], references: [id])

  @@index([createdAt])
  @@index([inboundOrderId])
  @@index([outboundOrderId])
  @@index([productId])
  @@index([type])
  @@index([warehouseId])
  @@map("inventory_logs")
}
```

### 5. InboundOrder 模型修改

```prisma
model InboundOrder {
  id              String             @id @default(cuid())
  inboundNo       String             @unique
  type            InboundType        @default(PURCHASE_IN)
  status          InboundStatus      @default(PENDING)
  purchaseOrderId String?
  supplierId      String?
  warehouseId     String             // 修改: 改为必填
  expectedDate    DateTime?
  actualDate      DateTime?
  totalAmount     Decimal            @db.Decimal(12, 2)
  note            String?
  createdAt       DateTime           @default(now())
  updatedAt       DateTime           @updatedAt
  items           InboundOrderItem[]
  purchaseOrder   PurchaseOrder?     @relation(fields: [purchaseOrderId], references: [id])
  supplier        Supplier?          @relation(fields: [supplierId], references: [id])
  inventoryLogs   InventoryLog[]
  warehouse       Warehouse          @relation(fields: [warehouseId], references: [id])

  @@index([status])
  @@index([warehouseId, status])
  @@index([createdAt])
  @@map("inbound_orders")
}
```

model OutboundOrder {
  id            String              @id @default(cuid())

  warehouseId   String

  items         OutboundOrderItem[]

  @@index([status])
  @@index([createdAt])
  @@index([warehouseId]) // 新增
  @@map("outbound_orders")
}
