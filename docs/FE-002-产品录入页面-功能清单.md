const basicInfoSchema = z.object({
  name: z.string().min(1, '产品名称不能为空'),
  brand: z.string().min(1, '品牌不能为空'),
  platform: z.enum(['Amazon', 'TikTok', '独立站', 'eBay', 'Shopee', 'Lazada', '其他']),
  purchasePrice: z.string().min(1, '采购价不能为空'),
  salePrice: z.string().min(1, '售价不能为空'),
  // ...
}).refine((data) => {
  if (parseFloat(data.salePrice) <= parseFloat(data.purchasePrice)) {
    return { message: '售价必须大于采购价', path: ['salePrice'] };
  }
  return true;
});
