### 1. ✅ batch-delete 添加软删除支持

**文件：** `src/app/api/products/batch-delete/route.ts`

**修改内容：**
- 将 `prisma.product.deleteMany()` 改为 `prisma.product.updateMany()` 
- 设置 `deletedAt: new Date()` 和 `status: 'INACTIVE'`
- 实现软删除，数据可恢复

**Schema 变更：**
- 在 `Product` 模型中添加 `deletedAt DateTime?` 字段
- 已运行迁移：`20260409135002_add_deleted_at_to_product`

**代码变更：**
```typescript
// 之前：硬删除
const result = await prisma.product.deleteMany({
  where: { id: { in: ids } },
});

// 现在：软删除
const result = await prisma.product.updateMany({
  where: { id: { in: ids } },
  data: {
    deletedAt: new Date(),
    status: 'INACTIVE',
  },
});
```

  return input
    .replace(/[%;'"\\]/g, '')  // 移除 SQL 特殊字符
    .replace(/--/g, '')        // 移除注释
    .replace(/\/\*/g, '')      // 移除块注释开始
    .replace(/\*\//g, '')      // 移除块注释结束

    .slice(0, 100);            // 限制长度

const validateEmail = (email: string): boolean => {
  if (!email) return true; // 允许为空
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  return emailRegex.test(email);
};

// 电话验证（支持国际格式）
const validatePhone = (phone: string): boolean => {
  if (!phone) return true; // 允许为空
  const phoneRegex = /^[\+]?[(]?[0-9]{1,4}[)]?[-\s\.]?[(]?[0-9]{1,4}[)]?[-\s\.]?[0-9]{1,9}$/;
  return phoneRegex.test(phone);
};
